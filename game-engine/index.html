import { Surface, Shape } from 'https://git.typable.dev/std/js/game.js';

class Player extends Surface {
	constructor(x, y) {
		super(x, y, new Shape.Circle(50));
	}
	render(g) {
		const { x, y, shape: { radius } } = this;
		g.beginPath();
		g.arc(x, y, radius, 2 * Math.PI, 0);
		g.stroke();
	}
}
